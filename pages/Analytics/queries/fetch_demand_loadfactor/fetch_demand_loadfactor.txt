SELECT 
    bm.customer_id,
    date_trunc('month', br.statement_date)::date AS time_period,
    biva.location_id,
    l.name AS location_description,
    bi.commodity AS utility_type,
    bi.bill_type,
    SUM(CASE WHEN c.name = 'demand_billblock' THEN c.value ELSE 0 END) AS demand,
    SUM(CASE WHEN c.name = 'load_factor_calc_billblock' THEN c.value ELSE NULL END) AS load_factor
FROM bill_management_v2.bill_metas bm
INNER JOIN bill_management_v2.bill_metas_bill_records bmbr ON bmbr.bill_meta_id = bm.id AND bmbr.status = 'active'
INNER JOIN bill_management_v2.bill_records br ON br.id = bmbr.bill_record_id
INNER JOIN bill_management_v2.bill_items bi ON bi.bill_record_id = br.id
INNER JOIN bill_management_v2.bill_items_virtual_accounts biva ON biva.bill_item_id = bi.id
INNER JOIN bill_management_v2.locations l ON l.id = biva.location_id
LEFT JOIN bill_management_v2.calculations c ON c.bill_item_id = bi.id
WHERE bm.deleted_at IS NULL AND bm.customer_ready IS NOT NULL
{{ CustomerSelect.selectedOptionValue ?
	`AND bm.customer_id = '${CustomerSelect.selectedOptionValue}'`
: "" }}
{{ DateSelect.selectedOptionValues?.filter(d => d.includes('-')).length > 0 ?
	`AND date_trunc('month', br.statement_date)::date IN (${DateSelect.selectedOptionValues.filter(d => d.includes('-')).map(d => `'${d}'`).join(",")})`
: "" }}

-- Utility Type
{{ UtilityTypeSelect.selectedOptionValues.length
? `AND bi.commodity IN ('${UtilityTypeSelect.selectedOptionValues.join("','")}')`
: "" }}

-- Bill Type
{{ BillTypeSelect.selectedOptionValue ?
	`AND bi.bill_type = '${BillTypeSelect.selectedOptionValue}'`
: "" }}

-- Location
{{ LocationSelect.selectedOptionValue ?
	`AND biva.location_id = ${LocationSelect.selectedOptionValue}`
: "" }}

GROUP BY bm.customer_id, date_trunc('month', br.statement_date)::date, biva.location_id, l.name, bi.commodity, bi.bill_type
ORDER BY l.name, date_trunc('month', br.statement_date)::date DESC
