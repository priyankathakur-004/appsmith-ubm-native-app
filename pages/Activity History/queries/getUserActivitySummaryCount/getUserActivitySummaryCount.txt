SELECT
    COUNT(*) AS total_count
FROM (
    SELECT
        u.id
    FROM bill_management_v2.activity_history ah
    LEFT JOIN bill_management_v2.bill_metas bm 
        ON bm.id = ah.bill_meta_id
    LEFT JOIN bill_management_v2.customers_search cs 
        ON cs.id = bm.customer_id 
    LEFT JOIN bill_management_v2.users u 
        ON u.id = ah.user_id
    WHERE
        1=1
        {{ users.selectedOptionValues && users.selectedOptionValues.length 
            ? `AND u.id IN (${users.selectedOptionValues.map(v => `'${v}'`).join(',')})`
            : ""
        }}
        {{ dateFrom.selectedDate && dateTo.selectedDate ? 
            `AND ah.created_at >= '${moment(dateFrom.selectedDate).startOf('day').format('YYYY-MM-DD HH:mm:ss')}'
             AND ah.created_at <= '${moment(dateTo.selectedDate).endOf('day').format('YYYY-MM-DD HH:mm:ss')}'`
            : ""
        }}
GROUP BY u.id
) AS grouped_users;
