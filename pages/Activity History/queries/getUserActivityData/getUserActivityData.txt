WITH paged_activity AS (
    SELECT *
    FROM bill_management_v2.activity_history
    WHERE
        user_id = {{ this.params.user_id }}
        {{ dateFrom.selectedDate && dateTo.selectedDate ? 
            `AND created_at BETWEEN
             '${moment(dateFrom.selectedDate).startOf('day').format('YYYY-MM-DD HH:mm:ss')}'
             AND
             '${moment(dateTo.selectedDate).endOf('day').format('YYYY-MM-DD HH:mm:ss')}'`
            : ""
        }}
    ORDER BY created_at DESC
    LIMIT {{ userActivityTable.pageSize }}
    OFFSET {{ (userActivityTable.pageNo - 1) * userActivityTable.pageSize }}
)

SELECT
    TO_CHAR(pa.created_at, 'MM-DD-YYYY HH24:MI') AS created_at,
    pa.bill_meta_id,
    STRING_AGG(DISTINCT bbm.batch_id::text, ', ') AS batch_ids,
		u.id as user_id,
    CONCAT(u.profile->>'first_name', ' ', u.profile->>'last_name') AS user_name,
    pa.text AS activity_text,
    cs.id AS customer_id,
    cs.name AS customer_name
FROM paged_activity pa
LEFT JOIN bill_management_v2.bill_metas bm 
    ON bm.id = pa.bill_meta_id
LEFT JOIN bill_management_v2.batches_bill_metas bbm 
    ON bbm.bill_meta_id = pa.bill_meta_id
LEFT JOIN bill_management_v2.customers_search cs 
    ON cs.id = bm.customer_id
LEFT JOIN bill_management_v2.users u 
    ON u.id = pa.user_id
GROUP BY
    pa.id,
    pa.created_at,
    pa.bill_meta_id,
    pa.text,
    cs.id,
    cs.name,
    u.profile,
		u.id
ORDER BY pa.created_at DESC;
