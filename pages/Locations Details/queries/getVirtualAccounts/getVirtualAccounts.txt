SELECT 
  v.pretty_name AS "Vendor",
  va.account_code AS "Billing ID",
  va.client_account AS "Service Account",
  va.meter_serial AS "Meter ID",
  va.commodity AS "Utility",
  va.bill_type AS "Type",
	va.last_bill AS "Last Bill"
FROM bill_management_v2.virtual_accounts_locations AS va
LEFT JOIN bill_management_v2.vendors AS v 
  ON v.code = va.vendor_code
WHERE va.customer_id = {{ appsmith.store.selectedCustomerId }}  AND va.location_id = {{ JSON.parse(appsmith.URL.queryParams.rowData).id }}
ORDER BY va.id LIMIT {{ pageSize }} OFFSET {{ (pageNo - 1) * pageSize }};
